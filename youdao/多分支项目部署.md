```sh
# 指定仓库目录-如：zuzuche/world、config_center
project=""
if [ $1 ];then
    project=$1
else
    echo "请输入要部署的仓库目录"
    exit 1
fi

# 部署的分支
branch="master"
if [ $2 ];then
    branch=$2
fi

# 多分支部署的根目录
release_dir="/data/htdocs_deploy/all_project/"

# 项目稳定版本的根目录
source_dir="/data/htdocs/stable_project/"

# 最终仓库发布的目录
release_project=$release_dir.$branch.'/'.$project
source_project=$source_dir.$project

# fetch 更新远程地址、清理无效 worktree
cd $source_project && git fetch -pa && git worktree prune

git worktree add --track -B $branch $release_project "$(git remote)/$branch"
```
