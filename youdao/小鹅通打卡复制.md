```javascript
var timer;
// 自动加载所有人的打卡信息
$(document).ready(function() {
  function loadAllData() {
    if ($(document).height() > $(window).height()) {
      timer = setTimeout(function() {
        $(window).scrollTop($(window).scrollTop() + $(window).height());
        loadAllData();
      }, 100);
     // 自动展开全文
    $("button.allArticle").each(function () {
        if($(this)[0].innerText == "全文") {
            $(this).click();
        }
    });
    }
  }

  loadAllData();
  // 设置 5s 后不再自动加载，可以自己滚动页面
  setTimeout(function() {
    clearTimeout(timer);
    alert("全部加载完毕!");
  }, 5000);
});

// 定义复制到剪切板
function copy() {
    // 复制内容
    var textToCopy = $("div.info_desc_wrap")[0].outerText + "\n\n--------";
    $("div.container-warp").each(function(i) {
        textToCopy += "\n\n" + $(this)[0].outerText;
    });
    console.log(textToCopy);
    navigator.clipboard.writeText(textToCopy).then(function() {
        alert('Text successfully copied to clipboard');
    }).catch(function(err) {
        alert('error!');
        console.error('Unable to copy text to clipboard', err);
    });
}

// 浏览器限制，必须要交互才可以复制信息到剪切板
$("div.footer_bt").prepend('<input id="copy" onclick="javascript:copy();" type="button" value="复制全部打卡记录">');
```
